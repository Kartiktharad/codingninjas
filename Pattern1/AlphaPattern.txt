Print the following pattern for the given N number of rows.
Pattern for N = 3
 A
 BB
 CCC
Input format :
Integer N (Total no. of rows)
Output format :
Pattern in N lines
Sample Input :
7
Sample Output :
A
BB
CCC
DDDD
EEEEE
FFFFFF
GGGGGGG

#include<iostream>
using namespace std;


int main(){

    /*  Read input as specified in the question.
     * Print output as specified in the question.
     */
  int n;
    cin>>n;
    for(int r=1;r<=n;r++)
    {
        for(int c=1;c<=r;c++)
            cout<<char(64+r);
        cout<<endl;
    }
    return 0;
}
